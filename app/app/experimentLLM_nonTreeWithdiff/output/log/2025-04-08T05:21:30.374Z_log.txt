// EngineHarness is responsible for managing Engine instances.
type EngineHarness struct {
	sync.RWMutex
	log           logging.Logger
	instances     []Engine
	started       atm.Bool
	faultDomain   *system.FaultDomain
	onDrpcFailure []func(context.Context, error)
}

// NewEngineHarness returns an initialized *EngineHarness.
func NewEngineHarness(log logging.Logger) *EngineHarness {
	return &amp;EngineHarness{
		log:       log,
		instances: make([]Engine, 0),
	}
}

// WithFaultDomain adds a fault domain to the EngineHarness.
func (h *EngineHarness) WithFaultDomain(fd *system.FaultDomain) *EngineHarness {
	h.faultDomain &#x3D; fd
	return h
}

// isStarted indicates whether the EngineHarness is in a running state.
func (h *EngineHarness) isStarted() bool {
	return h.started.Load()
}

// Instances safely returns harness&#x27; EngineInstances.
func (h *EngineHarness) Instances() []Engine {
	h.RLock()
	defer h.RUnlock()
	return h.instances
}

// FilterInstancesByRankSet returns harness&#x27; EngineInstances that match any
// of a list of ranks derived from provided rank set string.
func (h *EngineHarness) FilterInstancesByRankSet(ranks string) ([]Engine, error) {
	h.RLock()
	defer h.RUnlock()

	rankList, err :&#x3D; ranklist.ParseRanks(ranks)
	if err !&#x3D; nil {
		return nil, err
	}
	out :&#x3D; make([]Engine, 0)

	for _, i :&#x3D; range h.instances {
		r, err :&#x3D; i.GetRank()
		if err !&#x3D; nil {
			continue // no rank to check against
		}
		if r.InList(rankList) {
			out &#x3D; append(out, i)
		}
	}

	return out, nil
}

// AddInstance adds a new Engine instance to be managed.
func (h *EngineHarness) AddInstance(ei Engine) error {
	if h.isStarted() {
		return errors.New(&quot;can&#x27;t add instance to already-started harness&quot;)
	}

	h.Lock()
	defer h.Unlock()
	if indexSetter, ok :&#x3D; ei.(interface{ setIndex(uint32) }); ok {
		indexSetter.setIndex(uint32(len(h.instances)))
	}

	h.instances &#x3D; append(h.instances, ei)
	return nil
}

// OnDrpcFailure registers callbacks to be invoked on dRPC call failure.
func (h *EngineHarness) OnDrpcFailure(fns ...func(ctx context.Context, err error)) {
	h.Lock()
	defer h.Unlock()

	h.onDrpcFailure &#x3D; append(h.onDrpcFailure, fns...)
}

// CallDrpc calls the supplied dRPC method on a managed I/O Engine instance.
func (h *EngineHarness) CallDrpc(ctx context.Context, method drpc.Method, body proto.Message) (resp *drpc.Response, err error) {
	defer func() {
		if err &#x3D;&#x3D; nil {
			return
		}
		// If the context was canceled, don&#x27;t trigger callbacks.
		if errors.Cause(err) &#x3D;&#x3D; context.Canceled {
			return
		}
		// Don&#x27;t trigger callbacks for these errors which can happen when
		// things are still starting up.
		if err &#x3D;&#x3D; FaultHarnessNotStarted || err &#x3D;&#x3D; errEngineNotReady {
			return
		}

		h.log.Debugf(&quot;invoking dRPC failure handlers for %s&quot;, err)
		h.RLock()
		defer h.RUnlock()
		for _, fn :&#x3D; range h.onDrpcFailure {
			fn(ctx, err)
		}
	}()

	if !h.isStarted() {
		return nil, FaultHarnessNotStarted
	}

	// Iterate through the managed instances, looking for
	// the first one that is available to service the request.
	// If the request fails, that error will be returned.
	for _, i :&#x3D; range h.Instances() {
		resp, err &#x3D; i.CallDrpc(ctx, method, body)

		switch errors.Cause(err) {
		case errEngineNotReady, errDRPCNotReady, FaultDataPlaneNotStarted:
			continue
		default:
			return
		}
	}

	return
}

type dbLeader interface {
	IsLeader() bool
	ShutdownRaft() error
	ResignLeadership(error) error
}

// Start starts harness by setting up and starting dRPC before initiating
// configured instances&#x27; processing loops.
//
// Run until harness is shutdown.
func (h *EngineHarness) Start(ctx context.Context, db dbLeader, cfg *config.Server) error {
	if h.isStarted() {
		return errors.New(&quot;can&#x27;t start: harness already started&quot;)
	}

	if cfg &#x3D;&#x3D; nil {
		return errors.New(&quot;nil cfg supplied to Start()&quot;)
	}

	// Now we want to block any RPCs that might try to mess with storage
	// (format, firmware update, etc) before attempting to start I/O Engines
	// which are using the storage.
	h.started.SetTrue()
	defer h.started.SetFalse()

	for _, ei :&#x3D; range h.Instances() {
		ei.Run(ctx, cfg.RecreateSuperblocks)
	}

	h.OnDrpcFailure(func(_ context.Context, errIn error) {
		if !db.IsLeader() {
			return
		}

		switch errors.Cause(errIn) {
		case errDRPCNotReady, FaultDataPlaneNotStarted:
			break
		default:
			// Don&#x27;t shutdown on other failures which are
			// not related to dRPC communications.
			return
		}

		// If we cannot service a dRPC request on this node,
		// we should resign as leader in order to force a new
		// leader election.
		if err :&#x3D; db.ResignLeadership(errIn); err !&#x3D; nil {
			h.log.Errorf(&quot;failed to resign leadership after dRPC failure: %s&quot;, err)
		}
	})

	&lt;-ctx.Done()
	h.log.Debug(&quot;shutting down harness&quot;)

	return ctx.Err()
}

// readyRanks returns rank assignment of configured harness instances that are
// in a ready state. Rank assignments can be nil.
func (h *EngineHarness) readyRanks() []ranklist.Rank {
	h.RLock()
	defer h.RUnlock()

	ranks :&#x3D; make([]ranklist.Rank, 0)
	for idx, ei :&#x3D; range h.instances {
		if ei.IsReady() {
			rank, err :&#x3D; ei.GetRank()
			if err !&#x3D; nil {
				h.log.Errorf(&quot;instance %d: no rank (%s)&quot;, idx, err)
				continue
			}
			ranks &#x3D; append(ranks, rank)
		}
	}

	return ranks
}

func queryRank(reqRank uint32, engineRank ranklist.Rank) bool {
	rr :&#x3D; ranklist.Rank(reqRank)
	if rr.Equals(ranklist.NilRank) {
		return true
	}
	return rr.Equals(engineRank)

func (svc *ControlService) querySmdDevices(ctx context.Context, req *ctlpb.SmdQueryReq, resp *ctlpb.SmdQueryResp) error {
	for _, ei :&#x3D; range svc.harness.Instances() {
		if !ei.IsReady() {
			svc.log.Debugf(&quot;skipping not-ready instance %d&quot;, ei.Index())
			continue
		}

		engineRank, err :&#x3D; ei.GetRank()
		if err !&#x3D; nil {
			return err
		}
		if !queryRank(req.GetRank(), engineRank) {
			continue
		}

		rResp :&#x3D; new(ctlpb.SmdQueryResp_RankResp)
		rResp.Rank &#x3D; engineRank.Uint32()

		listDevsResp, err :&#x3D; ei.ListSmdDevices(ctx, new(ctlpb.SmdDevReq))
		if err !&#x3D; nil {
			return errors.Wrapf(err, &quot;rank %d&quot;, engineRank)
		}

		if len(listDevsResp.Devices) &#x3D;&#x3D; 0 {
			rResp.Devices &#x3D; nil
			resp.Ranks &#x3D; append(resp.Ranks, rResp)
			continue
		}

		// For each SmdDevice returned in list devs response, append a SmdDeviceWithHealth.
		for _, sd :&#x3D; range listDevsResp.Devices {
			rResp.Devices &#x3D; append(rResp.Devices, &amp;ctlpb.SmdQueryResp_SmdDeviceWithHealth{
				Details: sd,
			})
		}
		resp.Ranks &#x3D; append(resp.Ranks, rResp)

		if req.Uuid !&#x3D; &quot;&quot; {
			found :&#x3D; false
			for _, dev :&#x3D; range rResp.Devices {
				if dev.Details.Uuid &#x3D;&#x3D; req.Uuid {
					rResp.Devices &#x3D; []*ctlpb.SmdQueryResp_SmdDeviceWithHealth{dev}
					found &#x3D; true
					break
				}
			}
			if !found {
				rResp.Devices &#x3D; nil
			}
		}

		for _, dev :&#x3D; range rResp.Devices {
			state :&#x3D; dev.Details.DevState

			// skip health query if the device is not in a normal or faulty state
			if req.IncludeBioHealth {
				if state !&#x3D; ctlpb.NvmeDevState_NEW {
					health, err :&#x3D; ei.GetBioHealth(ctx, &amp;ctlpb.BioHealthReq{
						DevUuid: dev.Details.Uuid,
					})
					if err !&#x3D; nil {
						return errors.Wrapf(err, &quot;device %q, state %q&quot;,
							dev, state)
					}
					dev.Health &#x3D; health
					continue
				}
				svc.log.Debugf(&quot;skip fetching health stats on device %q in NEW state&quot;,
					dev, state)
			}
		}
	}

	return nil
}

// Split IDs in comma separated string and assign each token to relevant return list.
func extractReqIDs(log logging.Logger, ids string, addrs idMap, uuids idMap) error {
	tokens :&#x3D; strings.Split(ids, &quot;,&quot;)

	for _, token :&#x3D; range tokens {
		if addr, e :&#x3D; hardware.NewPCIAddress(token); e &#x3D;&#x3D; nil &amp;&amp; addr.IsVMDBackingAddress() {
			addrs[addr.String()] &#x3D; true
			continue
		}

		if uuid, e :&#x3D; uuid.Parse(token); e &#x3D;&#x3D; nil {
			uuids[uuid.String()] &#x3D; true
			continue
		}

		return errors.Errorf(&quot;req id entry %q is neither a valid vmd backing device pci &quot;+
			&quot;address or uuid&quot;, token)
	}

	return nil
}

// Map requested device IDs provided in comma-separated string to the engine that controls the given
// device. Device can be identified either by UUID or transport (PCI) address.
func (svc *ControlService) mapIDsToEngine(ctx context.Context, ids string, useTrAddr bool) (engineDevMap, error) {
	trAddrs :&#x3D; make(idMap)
	devUUIDs :&#x3D; make(idMap)
	matchAll :&#x3D; false

	if ids &#x3D;&#x3D; &quot;&quot; {
		// Selecting all is not supported unless using transport addresses.
		if !useTrAddr {
			return nil, errors.New(&quot;empty id string&quot;)
		}
		matchAll &#x3D; true
	} else {
		// Extract transport addresses and device UUIDs from IDs string.
		if err :&#x3D; extractReqIDs(svc.log, ids, trAddrs, devUUIDs); err !&#x3D; nil {
			return nil, err
		}
	}

	req :&#x3D; &amp;ctlpb.SmdQueryReq{Rank: uint32(ranklist.NilRank)}
	resp :&#x3D; new(ctlpb.SmdQueryResp)
	if err :&#x3D; svc.querySmdDevices(ctx, req, resp); err !&#x3D; nil {
		return nil, err
	}

	edm :&#x3D; make(engineDevMap)

	for _, rr :&#x3D; range resp.Ranks {
		engines, err :&#x3D; svc.harness.FilterInstancesByRankSet(fmt.Sprintf(&quot;%d&quot;, rr.Rank))
		if err !&#x3D; nil {
			return nil, err
		}
		if len(engines) &#x3D;&#x3D; 0 {
			return nil, errors.Errorf(&quot;failed to retrieve instance for rank %d&quot;,
				rr.Rank)
		}
		engine :&#x3D; engines[0]
		for _, dev :&#x3D; range rr.Devices {
			if dev &#x3D;&#x3D; nil {
				return nil, errors.New(&quot;nil device in smd query resp&quot;)
			}
			dds :&#x3D; dev.Details
			if dds &#x3D;&#x3D; nil {
				return nil, errors.New(&quot;device with nil details in smd query resp&quot;)
			}
			if dds.TrAddr &#x3D;&#x3D; &quot;&quot; {
				svc.log.Errorf(&quot;No transport address associated with device %s&quot;,
					dds.Uuid)
			}

			matchUUID :&#x3D; dds.Uuid !&#x3D; &quot;&quot; &amp;&amp; devUUIDs[dds.Uuid]

			// Where possible specify the TrAddr over UUID as there may be multiple
			// UUIDs mapping to the same TrAddr.
			if useTrAddr &amp;&amp; dds.TrAddr !&#x3D; &quot;&quot; {
				if matchAll || matchUUID || trAddrs[dds.TrAddr] {
					// If UUID matches, add by TrAddr rather than UUID which
					// should avoid duplicate UUID entries for the same TrAddr.
					edm.add(engine, devID{trAddr: dds.TrAddr})
					delete(trAddrs, dds.TrAddr)
					delete(devUUIDs, dds.Uuid)
					continue
				}
			}

			if matchUUID {
				// Only add UUID entry if TrAddr is not available for a device.
				edm.add(engine, devID{uuid: dds.Uuid})
				delete(devUUIDs, dds.Uuid)
			}
		}
	}

	// Check all input IDs have been matched.
	missingKeys :&#x3D; append(devUUIDs.Keys(), trAddrs.Keys()...)
	if len(missingKeys) &gt; 0 {
		return nil, errors.Errorf(&quot;ids requested but not found: %v&quot;, missingKeys)
	}

	return edm, nil
}

func sendManageReq(c context.Context, e Engine, m drpc.Method, b proto.Message) (*ctlpb.SmdManageResp_Result, error) {
	if !e.IsReady() {
		return &amp;ctlpb.SmdManageResp_Result{
			Status: daos.Unreachable.Int32(),
		}, nil
	}

	dResp, err :&#x3D; e.CallDrpc(c, m, b)
	if err !&#x3D; nil {
		return nil, errors.Wrap(err, &quot;call drpc&quot;)
	}

	mResp :&#x3D; new(ctlpb.DevManageResp)
	if err &#x3D; proto.Unmarshal(dResp.Body, mResp); err !&#x3D; nil {
		return nil, errors.Wrapf(err, &quot;unmarshal %T response&quot;, mResp)
	}

	return &amp;ctlpb.SmdManageResp_Result{
		Status: mResp.Status, Device: mResp.Device,
	}, nil
}

func addManageRespIDOnFail(log logging.Logger, res *ctlpb.SmdManageResp_Result, dev *devID) {
	if res &#x3D;&#x3D; nil || dev &#x3D;&#x3D; nil || res.Status &#x3D;&#x3D; 0 {
		return
	}

	log.Errorf(&quot;drpc returned status %q on dev %+v&quot;, daos.Status(res.Status), dev)
	if res.Device &#x3D;&#x3D; nil {
		// Populate id so failure can be mapped to a device.
		res.Device &#x3D; &amp;ctlpb.SmdDevice{
			TrAddr: dev.trAddr, Uuid: dev.uuid,
		}
	}
}

// Retry dev-replace requests as state propagation may take some time after set-faulty call has
// been made to manually trigger a faulty device state.
func replaceDevRetryBusy(ctx context.Context, log logging.Logger, e Engine, req proto.Message) (res *ctlpb.SmdManageResp_Result, err error) {
	for try :&#x3D; uint(0); try &lt; uint(maxDevReplaceRetries); try++ {
		res, err &#x3D; sendManageReq(ctx, e, drpc.MethodReplaceStorage, req)
		if err !&#x3D; nil {
			return
		}
		if daos.Status(res.Status) !&#x3D; daos.Busy {
			break
		}

		backoff :&#x3D; common.ExpBackoff(baseDevReplaceBackoff, uint64(try),
			uint64(maxDevReplaceBackoffFactor))
		log.Debugf(&quot;retrying dev-replace drpc request after %s&quot;, backoff)

		select {
		case &lt;-ctx.Done():
			err &#x3D; ctx.Err()
			return
		case &lt;-time.After(backoff):
		}
	}

	return
}

// SmdManage implements the method defined for the Management Service.
//
// Manage SMD devices.
func (svc *ControlService) SmdManage(ctx context.Context, req *ctlpb.SmdManageReq) (*ctlpb.SmdManageResp, error) {
	if !svc.harness.isStarted() {
		return nil, FaultHarnessNotStarted
	}
	if len(svc.harness.readyRanks()) &#x3D;&#x3D; 0 {
		return nil, FaultDataPlaneNotStarted
	}

	// Flag indicates whether Device-UUID can be replaced with its parent NVMe controller address.
	var useTrAddrInReq bool
	var ids string

	switch req.Op.(type) {
	case *ctlpb.SmdManageReq_Replace:
		ids &#x3D; req.GetReplace().OldDevUuid
	case *ctlpb.SmdManageReq_Faulty:
		ids &#x3D; req.GetFaulty().Uuid
	case *ctlpb.SmdManageReq_Led:
		useTrAddrInReq &#x3D; true
		ids &#x3D; req.GetLed().Ids
	default:
		return nil, errors.Errorf(&quot;Unrecognized operation in SmdManageReq: %+v&quot;, req.Op)
	}

	// Evaluate which engine(s) to send requests to.
	engineDevMap, err :&#x3D; svc.mapIDsToEngine(ctx, ids, useTrAddrInReq)
	if err !&#x3D; nil {
		return nil, errors.Wrap(err, &quot;mapping device identifiers to engine&quot;)
	}

	rankResps :&#x3D; []*ctlpb.SmdManageResp_RankResp{}

	for engine, devs :&#x3D; range engineDevMap {
		devResults :&#x3D; []*ctlpb.SmdManageResp_Result{}

		rank, err :&#x3D; engine.GetRank()
		if err !&#x3D; nil {
			return nil, errors.Wrap(err, &quot;retrieving engine rank&quot;)
		}

		msg :&#x3D; fmt.Sprintf(&quot;CtlSvc.SmdManage dispatch, rank %d: %%s req:%%+v\n&quot;, rank)

		// Extract request from oneof field and execute dRPC.
		switch req.Op.(type) {
		case *ctlpb.SmdManageReq_Replace:
			if len(devs) !&#x3D; 1 {
				return nil, errors.New(&quot;replace request expects only one device ID&quot;)
			}
			dReq :&#x3D; req.GetReplace()
			svc.log.Debugf(msg, &quot;dev-replace&quot;, dReq)
			devRes, err :&#x3D; replaceDevRetryBusy(ctx, svc.log, engine, dReq)
			if err !&#x3D; nil {
				return nil, errors.Wrap(err, msg)
			}
			addManageRespIDOnFail(svc.log, devRes, devs.getFirst())
			devResults &#x3D; append(devResults, devRes)
		case *ctlpb.SmdManageReq_Faulty:
			if len(devs) !&#x3D; 1 {
				return nil, errors.New(&quot;set-faulty request expects only one device ID&quot;)
			}
			dReq :&#x3D; req.GetFaulty()
			svc.log.Debugf(msg, &quot;set-faulty&quot;, dReq)
			devRes, err :&#x3D; sendManageReq(ctx, engine, drpc.MethodSetFaultyState, dReq)
			if err !&#x3D; nil {
				return nil, errors.Wrap(err, msg)
			}
			addManageRespIDOnFail(svc.log, devRes, devs.getFirst())
			devResults &#x3D; append(devResults, devRes)
		case *ctlpb.SmdManageReq_Led:
			if len(devs) &#x3D;&#x3D; 0 {
				// Operate on all devices by default.
				return nil, errors.New(&quot;led-manage request expects one or more IDs&quot;)
			}
			// Multiple addresses are supported in LED request.
			for _, dev :&#x3D; range devs {
				dReq :&#x3D; req.GetLed()
				// ID should by now have been resolved to a transport (PCI) address.
				if dev.trAddr &#x3D;&#x3D; &quot;&quot; {
					return nil, errors.Errorf(&quot;device uuid %s not resolved to a PCI address&quot;,
						dev.uuid)
				}
				dReq.Ids &#x3D; dev.trAddr
				svc.log.Debugf(msg, &quot;led-manage&quot;, dReq)
				devRes, err :&#x3D; sendManageReq(ctx, engine, drpc.MethodLedManage, dReq)
				if err !&#x3D; nil {
					return nil, errors.Wrap(err, msg)
				}
				addManageRespIDOnFail(svc.log, devRes, &amp;dev)
				devResults &#x3D; append(devResults, devRes)
			}
		default:
			return nil, errors.New(&quot;unexpected smd manage request type&quot;)
		}

		rankResps &#x3D; append(rankResps, &amp;ctlpb.SmdManageResp_RankResp{
			Rank: rank.Uint32(), Results: devResults,
		})
	}

	sort.Slice(rankResps, func(i, j int) bool {
		return rankResps[i].Rank &lt; rankResps[j].Rank
	})

	resp :&#x3D; &amp;ctlpb.SmdManageResp{Ranks: rankResps}

	return resp, nil
}
